import numpy as np
from collections import deque

"""
Transforms the input grid based on connectivity of azure (8) pixels.
Blue (1) and magenta (6) pixels act as barriers.
Azure (8) pixels that can reach the border of the grid (top, bottom, left, or right edge) by moving horizontally or vertically through other azure pixels remain azure (8).
Azure (8) pixels that are enclosed by barriers or other non-azure pixels and cannot reach the border are changed to orange (7).
All other pixels (including the barriers) retain their original color.
"""

def transform(input_grid):
    """
    Applies the flood-fill-from-border transformation rule to the input grid.

    Args:
        input_grid (list[list[int]]): A 2D list representing the input grid.

    Returns:
        list[list[int]]: A 2D list representing the transformed output grid.
    """
    # Convert input list of lists to numpy array for easier manipulation
    grid_np = np.array(input_grid, dtype=int)
    height, width = grid_np.shape

    # Initialize output_grid as a copy of the input grid
    output_grid = np.copy(grid_np)

    # Define barrier colors
    barrier_colors = {1, 6}

    # Create a boolean grid to mark azure pixels reachable from the border
    reachable = np.zeros_like(grid_np, dtype=bool)

    # Initialize a queue for Breadth-First Search (BFS)
    queue = deque()

    # Seed the BFS queue with all azure (8) pixels on the border
    for r in range(height):
        for c in range(width):
            # Check if the cell is on the border
            if r == 0 or r == height - 1 or c == 0 or c == width - 1:
                # Check if the border cell is azure and not already marked/queued
                if grid_np[r, c] == 8 and not reachable[r, c]:
                    queue.append((r, c))
                    reachable[r, c] = True

    # Define offsets for 4 neighbors (horizontal and vertical)
    neighbor_offsets = [(-1, 0), (1, 0), (0, -1), (0, 1)]

    # Perform BFS to find all reachable azure pixels
    while queue:
        r, c = queue.popleft()

        # Explore neighbors
        for dr, dc in neighbor_offsets:
            nr, nc = r + dr, c + dc

            # Check if the neighbor is within grid bounds
            if 0 <= nr < height and 0 <= nc < width:
                # Check if the neighbor is azure, not a barrier, and not already visited
                if grid_np[nr, nc] == 8 and not reachable[nr, nc]:
                    # Mark as reachable and add to the queue
                    reachable[nr, nc] = True
                    queue.append((nr, nc))

    # Iterate through the grid and update the output grid
    for r in range(height):
        for c in range(width):
            # If a pixel was originally azure (8) but is NOT reachable from the border
            if grid_np[r, c] == 8 and not reachable[r, c]:
                # Change it to orange (7) in the output grid
                output_grid[r, c] = 7
            # Otherwise, the pixel keeps its original color (already handled by the initial copy)

    # Convert back to list of lists for the expected output format
    return output_grid.tolist()
