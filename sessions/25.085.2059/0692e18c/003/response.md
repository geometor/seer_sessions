"""
Transforms a 3x3 input grid into a 9x9 output grid. 
The transformation rule involves identifying the unique non-white color (C) in the input grid.
A 3x3 'pattern_grid' (P) is generated by swapping the white color (0) and color C from the input grid.
The 9x9 output grid is conceptually divided into a 3x3 arrangement of 3x3 blocks.
Each block in the output grid corresponds to a pixel in the input grid.
If the input pixel input_grid[r, c] is white (0), the corresponding output block (starting at r*3, c*3) is filled with white (0).
If the input pixel input_grid[r, c] is the non-white color C, the corresponding output block is filled with the generated pattern_grid P.
"""

import numpy as np

def find_non_white_color(grid):
    """Finds the unique non-white color in the grid."""
    unique_colors = np.unique(grid)
    non_white_colors = unique_colors[unique_colors != 0]
    if len(non_white_colors) == 1:
        return non_white_colors[0]
    elif len(non_white_colors) == 0:
        # Handle case with no non-white colors (though not expected based on examples)
        # Default behavior might be needed, here we return None or raise error
        # For this specific problem, examples guarantee one non-white color.
        # Let's return a default or raise an error if assumptions change.
        # Returning 0 might be problematic if 0 is expected. Let's return None.
        return None 
    else:
        # Handle case with multiple non-white colors (not expected)
        # Raise error as the logic depends on a single non-white color
        raise ValueError("Input grid must contain exactly one non-white color.")

def generate_pattern_grid(grid, non_white_color):
    """Generates the pattern grid by swapping white (0) and the non_white_color."""
    if non_white_color is None:
         # If no non-white color, the pattern is just the grid itself? Or all zeros?
         # Based on the rule, if no non-white C, swapping isn't well-defined.
         # Let's assume the caller handles the 'None' case from find_non_white_color.
         # If called with None, maybe return an all-zero grid matching input shape.
         return np.zeros_like(grid)
         
    # Where grid is 0, put non_white_color. Where grid is non_white_color, put 0.
    # Need to be careful if other colors exist, but problem assumes only 0 and C.
    pattern = np.where(grid == 0, non_white_color, 0)
    return pattern

def transform(input_grid):
    # Convert input list of lists to numpy array for efficient operations
    input_array = np.array(input_grid, dtype=int)
    input_h, input_w = input_array.shape

    # Basic validation for expected 3x3 input
    if input_h != 3 or input_w != 3:
        # Handle unexpected input dimensions if necessary
        # For now, assume 3x3 based on problem description
        pass 

    # 1. Identify the unique non-white color (C)
    non_white_color_C = find_non_white_color(input_array)
    
    # If no non-white color found, the output should theoretically be all white?
    # Let's stick to the pattern generation assuming C exists, handle None C later if needed.
    # Based on examples, C always exists.

    # 2. Create the pattern_grid (P) by swapping 0 and C
    pattern_grid_P = generate_pattern_grid(input_array, non_white_color_C)

    # 3. Create the 9x9 output grid, initialized to white (0)
    output_h = input_h * 3
    output_w = input_w * 3
    output_grid = np.zeros((output_h, output_w), dtype=int)

    # 4. Iterate through each cell (r, c) of the 3x3 input grid
    for r in range(input_h):
        for c in range(input_w):
            # Get the color of the current input pixel
            input_pixel_color = input_array[r, c]

            # 5. Locate the corresponding 3x3 block in the output grid
            start_row = r * 3
            start_col = c * 3
            end_row = start_row + 3
            end_col = start_col + 3

            # 6. If the input cell contains the non-white color C
            if input_pixel_color == non_white_color_C:
                # Copy the pattern_grid P into this 3x3 output block
                output_grid[start_row:end_row, start_col:end_col] = pattern_grid_P
            # 7. If the input cell is white (0), leave the block as white (already initialized)
            # else: (input_pixel_color == 0) -> do nothing

    # 8. Return the completed 9x9 output grid as a list of lists
    return output_grid.tolist()
