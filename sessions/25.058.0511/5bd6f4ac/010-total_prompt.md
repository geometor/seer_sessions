    Returns:
        list of lists: The bottom-right 3x3 subgrid.
    """

    rows, cols = input_array.shape

    # Extract the bottom-right 3x3 subgrid
    output_array = input_array[rows-3:, cols-3:]

    # Convert the numpy array back to a list of lists
