import numpy as np

"""
Identifies the locations of the single blue (1) and single red (2) pixels in the input grid. 
Calculates the coordinates of the pixels lying on the straight line segment connecting the blue and red pixels using Bresenham's line algorithm. 
Initializes the output grid as a copy of the input grid.
Changes the color of all pixels on the calculated line segment, *except* for the start (blue pixel location) and end (red pixel location) points, to green (3) in the output grid.
"""

def find_pixel(grid, color):
    """Finds the coordinates (row, col) of the first pixel with the specified color."""
    coords = np.argwhere(grid == color)
    if coords.size > 0:
        return tuple(coords[0])
    return None

def bresenham_line(r0, c0, r1, c1):
    """
    Generates coordinates for a line from (r0, c0) to (r1, c1) using Bresenham's algorithm.
    Returns a list of (row, col) tuples.
    """
    points = []
    dr = abs(r1 - r0)
    dc = abs(c1 - c0)
    sr = 1 if r0 < r1 else -1
    sc = 1 if c0 < c1 else -1
    err = dr - dc

    while True:
        points.append((r0, c0))
        if r0 == r1 and c0 == c1:
            break
        e2 = 2 * err
        if e2 > -dc:
            err -= dc
            r0 += sr
        if e2 < dr:
            err += dr
            c0 += sc
            
    return points

def transform(input_grid):
    """
    Draws a green line between the blue and red pixels, excluding the endpoints.
    """
    # Initialize output_grid as a copy of the input
    output_grid = np.copy(input_grid)

    # Find the coordinates of the blue (1) and red (2) pixels
    blue_coords = find_pixel(output_grid, 1)
    red_coords = find_pixel(output_grid, 2)

    if blue_coords is None or red_coords is None:
        # Should not happen based on task description, but good practice
        return output_grid 

    # Unpack coordinates
    r_blue, c_blue = blue_coords
    r_red, c_red = red_coords

    # Generate the line path using Bresenham's algorithm
    line_path = bresenham_line(r_blue, c_blue, r_red, c_red)

    # Iterate through the path coordinates, skipping the first (blue) and last (red) points
    # The path must have at least 3 points for there to be intermediate points
    if len(line_path) > 2: 
        for r, c in line_path[1:-1]:
             # Ensure coordinates are within grid bounds (should be, but safety check)
            if 0 <= r < output_grid.shape[0] and 0 <= c < output_grid.shape[1]:
                # Change the color of the intermediate pixels to green (3)
                output_grid[r, c] = 3

    return output_grid
