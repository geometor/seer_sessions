def transform(input_grid):

    Removes white background (color 0) from the input grid.
    """
    output_grid = np.copy(input_grid)
    rows, cols = output_grid.shape

    # Iterate through all pixels
    for r in range(rows):
        for c in range(cols):
            # Check if the pixel is white (0)
