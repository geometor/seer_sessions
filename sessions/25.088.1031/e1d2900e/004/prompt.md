import numpy as np
from collections import deque

"""
Transforms the input grid based on the following rule:
A white pixel (0) is changed to blue (1) if and only if:
1. It is orthogonally adjacent to at least one red pixel (2) in the input grid.
2. It is reachable from at least one original blue pixel (1) in the input grid 
   via a path consisting only of orthogonally connected white pixels (0), 
   potentially including the target white pixel itself.
   (In other words, the white pixel belongs to a connected component of white 
   pixels that touches or contains an original blue pixel).

All other pixels (original red, original blue, and white pixels not meeting 
both conditions) remain unchanged.
"""

def transform(input_grid: np.ndarray) -> np.ndarray:
    """
    Applies the transformation rule to the input grid.

    Args:
        input_grid: A 2D numpy array representing the input grid.

    Returns:
        A 2D numpy array representing the transformed output grid.
    """
    height, width = input_grid.shape
    output_grid = np.copy(input_grid)
    
    # Define orthogonal neighbors delta coordinates
    neighbors = [(0, 1), (0, -1), (1, 0), (-1, 0)]

    # --- Step 1: Identify potential target pixels ---
    # Find all white pixels (0) that are orthogonally adjacent to red pixels (2).
    potential_targets = set()
    for r in range(height):
        for c in range(width):
            if input_grid[r, c] == 0: # If the pixel is white
                is_adj_to_red = False
                # Check all orthogonal neighbors
                for dr, dc in neighbors:
                    nr, nc = r + dr, c + dc
                    # Check if neighbor is within bounds and is red
                    if 0 <= nr < height and 0 <= nc < width and input_grid[nr, nc] == 2:
                        is_adj_to_red = True
                        break # Found a red neighbor, no need to check others
                if is_adj_to_red:
                    potential_targets.add((r, c)) # Mark this white pixel as a potential target

    # --- Step 2: Identify starting points for BFS/Flood Fill ---
    # Find all original blue pixels (1) which will act as seeds for the search.
    start_nodes = []
    for r in range(height):
        for c in range(width):
            if input_grid[r, c] == 1:
                start_nodes.append((r, c))

    # --- Step 3: Perform BFS to find reachable white pixels from blue pixels ---
    # Use Breadth-First Search (BFS) starting from original blue pixels (1)
    # to find all connected white pixels (0).
    # Keep track of which potential targets are encountered during this search.
    
    queue = deque()
    visited = set() # Keep track of visited pixels (blue and white) to avoid cycles and redundant work
    confirmed_targets = set() # Store white pixels that are both potential targets and reachable from blue

    # Initialize queue and visited set with original blue pixels
    for r, c in start_nodes:
        if (r, c) not in visited:
             queue.append((r, c))
             visited.add((r, c))

    # Start BFS. The search expands through connected white pixels.
    while queue:
        r, c = queue.popleft() # Get the next pixel coordinates from the queue

        # Explore orthogonal neighbors of the current pixel
        for dr, dc in neighbors:
            nr, nc = r + dr, c + dc

            # Check if the neighbor is within grid bounds
            if 0 <= nr < height and 0 <= nc < width:
                # Check if the neighbor is white (0) and has not been visited yet
                if input_grid[nr, nc] == 0 and (nr, nc) not in visited:
                    visited.add((nr, nc)) # Mark as visited
                    queue.append((nr, nc)) # Add to the queue to explore its neighbors later
                    
                    # Crucially, check if this newly reached white pixel is one of the potential targets
                    if (nr, nc) in potential_targets:
                        # If yes, it meets both conditions (adjacent to red AND reachable from blue via white path)
                        confirmed_targets.add((nr, nc)) 
                
                # We only expand the search *through* white pixels. 
                # Blue pixels start the search but don't propagate it further once reached again,
                # as they are already handled by the initial seeding or added to visited.
                # Red pixels block the search.

    # --- Step 4: Apply transformation to confirmed targets ---
    # Modify the output grid: change the color of confirmed target pixels to blue (1).
    for r, c in confirmed_targets:
        output_grid[r, c] = 1 # Change color to blue

    # Return the modified grid
    return output_grid
