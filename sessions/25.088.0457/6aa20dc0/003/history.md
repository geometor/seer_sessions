import numpy as np
from collections import Counter

"""
The transformation identifies three distinct non-background objects in the input grid. 
One object acts as a 'key', and the other two act as 'targets'. 
The 'key' object is identified as the one containing a color not present in either of the 'target' objects. 
The 'fill color' is this unique color from the 'key' object.
A bounding box is determined that minimally encloses both 'target' objects.
The output grid is generated by filling all background-colored pixels within this bounding box with the 'fill color'. 
The original objects (key and targets) remain unchanged in their positions.
"""

def find_objects(grid, background_color):
    """Finds connected components of non-background colors."""
    rows, cols = grid.shape
    visited = np.zeros_like(grid, dtype=bool)
    objects = []

    for r in range(rows):
        for c in range(cols):
            if grid[r, c] != background_color and not visited[r, c]:
                obj_pixels = set()
                obj_colors = set()
                q = [(r, c)]
                visited[r, c] = True
                
                while q:
                    row, col = q.pop(0)
                    obj_pixels.add((row, col))
                    obj_colors.add(grid[row, col])
                    
                    # Check 4 neighbors (up, down, left, right)
                    for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                        nr, nc = row + dr, col + dc
                        if 0 <= nr < rows and 0 <= nc < cols and \
                           not visited[nr, nc] and grid[nr, nc] != background_color:
                            visited[nr, nc] = True
                            q.append((nr, nc))
                            
                if obj_pixels:
                    objects.append({'pixels': obj_pixels, 'colors': obj_colors})
    return objects

def get_bounding_box(pixels):
    """Calculates the bounding box for a set of pixels."""
    if not pixels:
        return None
    min_r = min(r for r, c in pixels)
    max_r = max(r for r, c in pixels)
    min_c = min(c for r, c in pixels)
    max_c = max(c for r, c in pixels)
    return min_r, max_r, min_c, max_c

def classify_objects_and_find_fill_color(objects):
    """
    Classifies objects into key and targets and determines the fill color.
    The key object has a color not present in the other two.
    The fill color is that unique color.
    Assumes exactly 3 objects based on examples.
    """
    if len(objects) != 3:
        # Handle cases differently if needed, but examples show 3
        print(f"Warning: Expected 3 objects, found {len(objects)}")
        # Basic fallback or error handling could go here.
        # For now, let's try to proceed assuming the logic might still apply or 
        # return None if it definitely won't work.
        if len(objects) < 3: return None, None, None 
        # If more than 3, maybe try combinations? Sticking to 3 for now.

    obj_colors = [o['colors'] for o in objects]

    for i in range(3):
        key_candidate_idx = i
        target_indices = [j for j in range(3) if j != i]
        
        key_colors = obj_colors[key_candidate_idx]
        target1_colors = obj_colors[target_indices[0]]
        target2_colors = obj_colors[target_indices[1]]
        
        # Find colors in key_candidate not in either target
        unique_key_colors = key_colors - target1_colors - target2_colors
        
        if len(unique_key_colors) == 1:
            fill_color = list(unique_key_colors)[0]
            key_object = objects[key_candidate_idx]
            target1 = objects[target_indices[0]]
            target2 = objects[target_indices[1]]
            return key_object, target1, target2, fill_color
            
    # If no unique key configuration found

    return None, None, None, None 

def calculate_combined_bounding_box(box1, box2):
    """Calculates the minimal bounding box enclosing two bounding boxes."""
    min_r = min(box1[0], box2[0])
    max_r = max(box1[1], box2[1])
    min_c = min(box1[2], box2[2])
    max_c = max(box1[3], box2[3])
    return min_r, max_r, min_c, max_c

def transform(input_grid):
    """
    Transforms the input grid by identifying key and target objects, determining
    a fill color from the key, calculating the bounding box of the targets,
    and filling the background within that box with the fill color.
    """
    input_grid = np.array(input_grid)
    
    # 1. Determine the background color
    counts = Counter(input_grid.flatten())
    background_color = counts.most_common(1)[0][0]
    
    # 2. Identify all distinct objects
    objects = find_objects(input_grid, background_color)
    
    # Add bounding box info to each object
    for obj in objects:
        obj['bbox'] = get_bounding_box(obj['pixels'])

    # 3. Distinguish key and target objects and determine fill color
    key_object, target1, target2, fill_color = classify_objects_and_find_fill_color(objects)

    if fill_color is None or target1 is None or target2 is None:
         # Handle cases where classification failed - perhaps return input?
         print("Error: Failed to classify objects or find fill color.")
         return input_grid.tolist() # Or raise an error

    # 4. Calculate the combined bounding box for target objects
    bbox1 = target1['bbox']
    bbox2 = target2['bbox']
    combined_bb = calculate_combined_bounding_box(bbox1, bbox2)
    min_r, max_r, min_c, max_c = combined_bb

    # 5. Create the output grid (copy of input)
    output_grid = np.copy(input_grid)

    # 6. Fill background pixels within the combined bounding box
    for r in range(min_r, max_r + 1):
        for c in range(min_c, max_c + 1):
            # Check if the pixel in the *original* grid is background
            if input_grid[r, c] == background_color:
                # Change the pixel in the *output* grid
                output_grid[r, c] = fill_color

    return output_grid.tolist()
