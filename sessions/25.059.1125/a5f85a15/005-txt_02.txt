import numpy as np

def describe_grid(grid):
    rows, cols = grid.shape
    description = {
        'shape': (rows, cols),
        'pixels': {}
    }
    for r in range(rows):
        for c in range(cols):
            value = grid[r, c]
            if value not in description['pixels']:
                description['pixels'][value] = []
            description['pixels'][value].append((r, c))
    return description
