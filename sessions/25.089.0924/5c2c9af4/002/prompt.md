import numpy as np

"""
Transformation rule:
1. Find all non-background (non-zero) pixels in the input grid.
2. If no non-zero pixels exist, return the input grid unchanged.
3. Identify the color (`C`) of these non-zero pixels (assuming they are all the same color).
4. Determine the minimum bounding box enclosing all pixels of color `C`.
5. Create an output grid by copying the input grid.
6. Starting with an expansion distance `k = 2`, draw concentric rectangular frames around the bounding box using color `C`.
7. Each subsequent frame increases the expansion distance by 2 (k = 4, 6, 8, ...).
8. Frames are drawn by calculating the expanded bounding box coordinates (min_row - k, max_row + k, min_col - k, max_col + k).
9. Only draw pixels that fall within the grid boundaries.
10. Continue drawing frames with increasing `k` until the expanded frame is entirely outside the grid boundaries.
11. Return the modified output grid.
"""

def find_non_background_pixels(grid):
    """Finds coordinates and the single color of non-background pixels."""
    non_zero_coords = np.argwhere(grid != 0)
    if len(non_zero_coords) == 0:
        return None, None, None # No non-background pixels found
    
    # Assuming all non-background pixels have the same color
    color = grid[non_zero_coords[0][0], non_zero_coords[0][1]]
    
    # Separate rows and columns
    rows = non_zero_coords[:, 0]
    cols = non_zero_coords[:, 1]
    
    return rows, cols, color

def calculate_bounding_box(rows, cols):
    """Calculates the min/max row/col for a set of coordinates."""
    if rows is None or len(rows) == 0:
        return None # Should not happen if find_non_background_pixels checks first
        
    min_r = np.min(rows)
    max_r = np.max(rows)
    min_c = np.min(cols)
    max_c = np.max(cols)
    
    return min_r, max_r, min_c, max_c

def transform(input_grid):
    """
    Applies the transformation rule to the input grid.
    Draws concentric frames with spacing 2 around the bounding box of non-zero pixels.
    """
    input_np = np.array(input_grid, dtype=int)
    height, width = input_np.shape
    
    # Find non-background pixels and their color
    rows, cols, color = find_non_background_pixels(input_np)
    
    # If no non-background pixels, return the original grid
    if rows is None:
        return input_grid
        
    # Calculate the bounding box of the initial pixels
    min_r, max_r, min_c, max_c = calculate_bounding_box(rows, cols)
    
    # Initialize the output grid as a copy of the input
    output_grid = np.copy(input_np)
    
    # Start drawing frames with expansion k=2, incrementing by 2
    k = 2
    while True:
        # Calculate coordinates of the expanded box for the current k
        top_row = min_r - k
        left_col = min_c - k
        bottom_row = max_r + k
        right_col = max_c + k
        
        # Check if the entire expanded box is outside the grid boundaries
        # If the bottom is above the top edge OR the top is below the bottom edge OR
        # the right is left of the left edge OR the left is right of the right edge
        if bottom_row < 0 or top_row >= height or right_col < 0 or left_col >= width:
            # The box is completely outside, no more frames can be drawn
            break
            
        frame_drawn = False # Flag to check if any part of the frame was within bounds
            
        # Draw the frame border, clipping to grid boundaries
        
        # Top border
        if 0 <= top_row < height:
            for c in range(max(0, left_col), min(width, right_col + 1)):
                 output_grid[top_row, c] = color
                 frame_drawn = True
                 
        # Bottom border
        if 0 <= bottom_row < height:
             for c in range(max(0, left_col), min(width, right_col + 1)):
                 output_grid[bottom_row, c] = color
                 frame_drawn = True

        # Left border (excluding corners already drawn)
        if 0 <= left_col < width:
            for r in range(max(0, top_row + 1), min(height, bottom_row)):
                 output_grid[r, left_col] = color
                 frame_drawn = True

        # Right border (excluding corners already drawn)
        if 0 <= right_col < width:
            for r in range(max(0, top_row + 1), min(height, bottom_row)):
                 output_grid[r, right_col] = color
                 frame_drawn = True

        # If no part of the frame could be drawn (e.g., k grew very large), stop.
        # This is an alternative break condition, the first one should catch most cases.
        if not frame_drawn and k > max(height, width): # Heuristic break
             break

        # Increment the expansion distance for the next frame
        k += 2
        
    # Return the final grid (converted back to list of lists if necessary for ARC format)
    return output_grid.tolist()
