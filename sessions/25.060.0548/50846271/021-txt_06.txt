#check if hypothesis is correct, leftmost red becomes azure
def check_leftmost(red, azure):
    for a in azure:
        found = False
        for r in red:
            if (a[0] == r[0] and a[1] == r[1]):
                found = True

        if not found:
            return False
    return True
