def compare_grids(grid1, grid2):
    """Compares two grids and returns a list of differences."""
    differences = []
    if len(grid1) != len(grid2):
        differences.append(f"Grid height mismatch: {len(grid1)} vs {len(grid2)}")
        return differences
    if len(grid1[0]) != len(grid2[0]):
        differences.append(f"Grid width mismatch: {len(grid1[0])} vs {len(grid2[0])}")
        return differences
    
    for i in range(len(grid1)):
        for j in range(len(grid1[0])):
            if grid1[i][j] != grid2[i][j]:
                differences.append(f"Pixel mismatch at ({i}, {j}): {grid1[i][j]} vs {grid2[i][j]}")
    return differences

def test_transform_on_all_examples(task_data, transform_func):
    results = {}
    for example_index, example in enumerate(task_data['train']):
        input_grid = example['input']
        expected_output_grid = example['output']
        transformed_grid = transform_func(input_grid)
        differences = compare_grids(transformed_grid, expected_output_grid)

        results[f'example_{example_index}'] = {
            'success': len(differences) == 0,
            'differences': differences
        }
    return results

def transform(input_grid):
    """
    Rotates the input grid 90 degrees counter-clockwise.
    """
    # Convert the input grid to a NumPy array for easier manipulation.
    input_array = np.array(input_grid)

    # Rotate the array 90 degrees counter-clockwise.
    output_array = np.rot90(input_array)

    # Convert the rotated array back to a list.
    output_grid = output_array.tolist()

    return output_grid
