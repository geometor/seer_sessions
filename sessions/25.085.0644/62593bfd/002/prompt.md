import numpy as np

def find_bounding_box(grid, bg_color):
    """
    Finds the minimum bounding box containing all non-background pixels.

    Args:
        grid (np.array): The input grid.
        bg_color (int): The background color.

    Returns:
        tuple: (min_row, max_row, min_col, max_col) or None if no non-bg pixels.
    """
    non_bg_coords = np.argwhere(grid != bg_color)
    if non_bg_coords.size == 0:
        return None
    min_row = non_bg_coords[:, 0].min()
    max_row = non_bg_coords[:, 0].max()
    min_col = non_bg_coords[:, 1].min()
    max_col = non_bg_coords[:, 1].max()
    return min_row, max_row, min_col, max_col

def transform(input_grid):
    """
    Identifies the background color (assumed to be the color of the top-left pixel).
    Finds the minimum bounding box enclosing all non-background pixels in the input grid.
    Creates an output grid of the same dimensions as the input, filled with the background color.
    Copies the content of the input grid within the calculated bounding box (including any background pixels within that box) 
    to the top-left corner (0,0) of the output grid.
    """
    # Convert input list of lists to numpy array for efficient processing
    grid = np.array(input_grid, dtype=int)
    height, width = grid.shape

    # Determine background color - assuming the top-left pixel is representative
    # or the most frequent color if top-left is part of an object near the edge.
    # Let's stick with top-left for simplicity based on examples.
    bg_color = grid[0, 0]

    # Find the bounding box of non-background elements
    bbox = find_bounding_box(grid, bg_color)

    # Initialize the output grid with the background color
    output_grid = np.full_like(grid, bg_color)

    # If a bounding box was found (i.e., non-background pixels exist)
    if bbox:
        min_row, max_row, min_col, max_col = bbox

        # Extract the subgrid defined by the bounding box from the input
        sub_grid = grid[min_row : max_row + 1, min_col : max_col + 1]
        sub_height, sub_width = sub_grid.shape

        # Check if the subgrid fits within the output grid dimensions (it should by definition)
        if sub_height <= height and sub_width <= width:
            # Copy the extracted subgrid to the top-left corner of the output grid
            output_grid[0:sub_height, 0:sub_width] = sub_grid
        else:
            # This case should not happen if output_grid has the same dims as input
            # but added as a safeguard. Copy only the part that fits.
            h_fit = min(sub_height, height)
            w_fit = min(sub_width, width)
            output_grid[0:h_fit, 0:w_fit] = sub_grid[0:h_fit, 0:w_fit]

    # Convert the result back to a list of lists as required by the ARC format
    return output_grid.tolist()
