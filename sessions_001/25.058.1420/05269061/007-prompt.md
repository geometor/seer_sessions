"""
The transformation rule involves identifying a 3x3 seed pattern in the top-left corner of the input grid and creating a dynamic seed by replacing the 0 values based on their position and a repeating sequence of non-zero colors from the input's first row. The output grid is then generated by tiling this dynamic seed.
"""

import numpy as np

def get_seed(grid):
    """Extracts the 3x3 seed pattern from the input grid."""
    return grid[:3, :3].copy()

def get_repeating_sequence(grid):
    """Gets the repeating sequence of non-zero colors from the first row."""
    first_row = grid[0]
    return first_row[first_row != 0]

def create_dynamic_seed(seed, sequence):
    """
    Creates a dynamic seed by replacing 0s in the seed with values from the
    repeating sequence, based on cell position.
    """
    dynamic_seed = seed.copy()
    for i in range(3):
        for j in range(3):
            if dynamic_seed[i, j] == 0:
                index = (i + j) % len(sequence)
                dynamic_seed[i, j] = sequence[index]
    return dynamic_seed

def transform(input_grid):
    """Transforms the input grid according to the defined rule."""
    input_grid = np.array(input_grid)
    rows, cols = input_grid.shape
    output_grid = np.zeros_like(input_grid)

    # 1. Identify the Seed.
    seed = get_seed(input_grid)

    # 2. Get Repeating Sequence.
    sequence = get_repeating_sequence(input_grid)

    # 3. Create Dynamic Seed.
    dynamic_seed = create_dynamic_seed(seed, sequence)

    # 4. Tile the Dynamic Seed.
    for i in range(rows):
        for j in range(cols):
            output_grid[i, j] = dynamic_seed[i % 3, j % 3]

    return output_grid.tolist()
