import numpy as np

def report(task, results):
    print(f"Task: {task}")
    for i, (input_grid, expected_output, actual_output) in enumerate(results):
        
        print(f"  Example {i + 1}:")
        print(f"    Input:\n{np.array(input_grid)}")
        print(f"    Expected Output:\n{np.array(expected_output)}")
        print(f"    Actual Output:\n{np.array(actual_output)}")

        if np.array_equal(np.array(expected_output), np.array(actual_output)):
          print("PASS")
        else:
          print("FAIL")
