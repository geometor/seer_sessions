"""
The input grid is expanded to create the output grid. The output grid's dimensions are double the input grid's dimensions in both height and width. The contents of the output grid are generated by mirroring the input grid both horizontally and vertically.
"""

import numpy as np

def transform(input_grid):
    # Get dimensions of the input grid
    input_rows, input_cols = input_grid.shape

    # Calculate dimensions of the output grid
    output_rows = input_rows * 2
    output_cols = input_cols * 2

    # Initialize the output grid with zeros
    output_grid = np.zeros((output_rows, output_cols), dtype=int)

    # Iterate through the input grid and replicate it in the output grid
    for i in range(input_rows):
        for j in range(input_cols):
            # Copy each pixel to its mirrored positions
            output_grid[i, j] = input_grid[i, j]
            output_grid[i, output_cols - 1 - j] = input_grid[i, j]
            output_grid[output_rows - 1 - i, j] = input_grid[i, j]
            output_grid[output_rows - 1 - i, output_cols - 1 - j] = input_grid[i, j]

    return output_grid
