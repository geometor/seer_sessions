"""
Recolors pixels within a gray (5) object based on their local connectivity.
Gray pixels with exactly one non-diagonal gray neighbor become red (2).
Gray pixels with two or more non-diagonal gray neighbors become azure (8).
White (0) background pixels remain unchanged.
"""

import numpy as np

def transform(input_grid):
    """
    Transforms the input grid by recoloring gray pixels based on neighbor count.

    Args:
        input_grid (np.array): A 2D numpy array representing the input grid.

    Returns:
        np.array: A 2D numpy array representing the transformed output grid.
    """
    # Get the dimensions of the input grid
    height, width = input_grid.shape

    # Initialize the output grid as a copy of the input grid
    # This handles the background (white) pixels staying the same.
    output_grid = np.copy(input_grid)

    # Iterate through each pixel in the input grid
    for r in range(height):
        for c in range(width):
            # Check if the current pixel is gray (5)
            if input_grid[r, c] == 5:
                gray_neighbor_count = 0
                # Define the relative coordinates for non-diagonal neighbors
                neighbors = [(0, 1), (0, -1), (1, 0), (-1, 0)] # Right, Left, Down, Up

                # Check each non-diagonal neighbor
                for dr, dc in neighbors:
                    nr, nc = r + dr, c + dc

                    # Check if the neighbor is within the grid boundaries
                    if 0 <= nr < height and 0 <= nc < width:
                        # Check if the neighbor is also gray (5)
                        if input_grid[nr, nc] == 5:
                            gray_neighbor_count += 1

                # Apply the coloring rule based on the neighbor count
                if gray_neighbor_count == 1:
                    output_grid[r, c] = 2 # Change to red
                elif gray_neighbor_count >= 2:
                    output_grid[r, c] = 8 # Change to azure
                # Note: If gray_neighbor_count is 0 (isolated gray pixel), 
                # it remains gray in the output_grid based on the copy. 
                # However, the examples don't show isolated gray pixels. 
                # If needed, a rule for count == 0 could be added.

    return output_grid
